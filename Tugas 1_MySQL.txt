//Membuat table pegawai
CREATE TABLE pegawai (
    id INT PRIMARY KEY,
    nip CHAR(5) NOT NULL UNIQUE,
    nama VARCHAR(30) NOT NULL,
    gender ENUM('L','P') NOT NULL,
    umur INT NOT NULL,
    berat_badan FLOAT
);

//Menampilkan table pegawai
SHOW TABLES;

//Menampilkan deskripsi tabel pegawai yang sudah dibuat
DESC pegawai;
